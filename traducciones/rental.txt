prontera,132,190,7	script	Universal Rental	726,{
	mes "[ ^2e9c99Universal Rental^000000 ]";
	mes "Hi!";
	mes "With me you can get a Cart, Falcon or a PecoPeco, depending on the job and if you meet the requirements that this entails.";
	next;
	mes "[ ^2e9c99Universal Rental^000000 ]";
	mes "What do you want to do?";
	next;
	menu "~ Rent a PecoPeco, Falcon o cart.", M1, "~ ^009900Rent premium carts^000000.", M2;
	
M1: if (ismounting()) {
		dispbottom "#IRO: You must get off your mount to use this service.", 0xffff00;
		end;
	} else if (((eaclass()&EAJ_THIRDMASK)==EAJ_RANGER) && !countitem(6124)) {
		if (!checkfalcon() && getskilllv("HT_FALCON") && !checkwug()) {
			if(select(" ~ Falcon: ~ Warg")==1) setfalcon;
			else getitem 6124,1; //Wolf's_Flute
		} else getitem 6124,1; //Wolf's_Flute
	} else if (((eaclass()&EAJ_THIRDMASK)==EAJ_MECHANIC) && !checkcart() && getskilllv("MC_PUSHCART")) {
		if (!checkmadogear() && getskilllv("NC_MADOLICENCE")) {
			if(select(" ~ Cart: ~ Mado")==1) setcart;
			else setmadogear;
		} else setcart;
	} else if (!checkcart() && getskilllv("MC_PUSHCART")) setcart;
	else if (!checkfalcon() && getskilllv("HT_FALCON") && !checkwug()) setfalcon;
	else if (!checkriding() && !checkdragon() && getskilllv("KN_RIDING")) {
		if ((eaclass()&EAJ_THIRDMASK)==EAJ_RUNE_KNIGHT) setdragon;
		else setriding;
	} else if (!checkmadogear() && getskilllv("NC_MADOLICENCE")) setmadogear;
	else {
		dispbottom "#IRO: You don't meet the requirements to have this service.", 0xffff00;
		end;
	}
	specialeffect2 EF_TEIHIT3;
	end;
	

M2: 	mes "[ ^2e9c99Universal Rental^000000 ]";
	mes "The ^800080premium carts^000000 are 7 new carts designs for Merchant class.";
	next;
	mes "[ ^2e9c99Universal Rental^000000 ]";
	mes "The cost to unlock all the ^800080premium carts^000000 designs in your account is 20,000 ^FF6133Cash points^000000.";
	next;
	switch(select("~ ^777777See designs^000000.", (!#MCDECOR2PAID) ? "~ ^009900Unlock premium carts^000000.":"", (#MCDECOR2PAID) ? "~ ^1E90FFSkill Change Cart 2^000000.":"")) {
	case 1:
		// The picture is read from data\texture\Ã¬Å“Â Ã¬Â â‚¬Ã¬ÂÂ¸Ã­â€žÂ°Ã­Å½ËœÃ¬ÂÂ´Ã¬Å Â¤\illust
		// cutin file name (must be *.bmp)
		cutin "carts_new",4;
		mes "[ ^2e9c99Universal Rental^000000 ]";
		mes "These are the ^800080premium carts^000000.";
		close2;
		cutin "",255;
		break;
		
	case 2:
       	 	if ( !#MCDECOR2PAID ) {
		mes "[ ^2e9c99Universal Rental^000000 ]";
        	mes "I'll need you give me 20,000 ^FF6133Cash points^000000 to unlock all the ^800080premium account^000000 in your account.";
         	mes "Actual State: "+ #CASHPOINTS + " ^FF6133Cash points^000000.";
		mes "Do you have them ready?";
        	next;
        	if (select("~ ^009900Give Cash points^000000.:~ ^FF0000No, not yet^000000.")==2)close;
           	   if ( #CASHPOINTS < 20000 ) {
			mes "[ ^00BFFFUniversal Rental^000000^000000 ]";
			mes "~ ^FF0000Sorry, you don't have what i need to enjoy this service^000000.";
                	close;
            	}
       	 }
		mes "[ ^2e9c99Universal Rental^000000 ]";
		mes "^009900¡Listo!^000000";
		mes "Press (Alt + S) and you'll see your skill tree, on this, you gonna find a skill called *^1E90FFChange Cart 2^000000* With it you can choose all the ^800080Premium carts^000000.";
		next;
		mes "[ ^2e9c99Universal Rental^000000 ]";
		mes "Remember, now all the characters on this account that are of the Merchant Class, talking with me, will be able to enjoy this service and no need to pay anything else.";
        	close2;
        	skill 2544,1,SKILL_PERM_GRANT;
        	#CASHPOINTS -= 20000;
        	#MCDECOR2PAID = true;
       	 	break;	
	case 3:
		mes "[ ^2e9c99Universal Rental^000000 ]";
		mes "^009900It's ready!^000000";
		mes "Press (Alt + S) and you'll see your skill tree, on this, you gonna find a skill called *^1E90FFChange Cart 2^000000* With it you can choose all the ^800080carros premium^000000.";
		close2;
		skill 2544,1,SKILL_PERM_GRANT;
		break;
	}
	end;

		waitingroom "[ Universal Rental ]",0;
end;
